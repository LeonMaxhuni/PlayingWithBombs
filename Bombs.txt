import javax.swing.JPanel;

public class Bombs extends JPanel {

    int blockSize = 64;

    int currentPlayerX, currentPlayerY;
    int previousPlayerX, previousPlayerY = 100;

    public boolean firstBomb, secondBomb, thirdBomb = false;
    int bomb1X, bomb1Y, bomb2X, bomb2Y, bomb3X, bomb3Y;

    int count = 0;
    int maxBombs = 3;

    long end, end2, end3;
    MyMap mappy2;

    public Bombs() {

    }

    public void setLocation(int playerX, int playerY) {

        currentPlayerX = playerX;
        currentPlayerY = playerY;

        if(count < maxBombs && (previousPlayerX != currentPlayerX || previousPlayerY != currentPlayerY)) {

            previousPlayerX = currentPlayerX;
            previousPlayerY = currentPlayerY;

            if (secondBomb == true && thirdBomb == false) {
                thirdBomb = true;
                count++;
                bomb3X = previousPlayerX;
                bomb3Y = previousPlayerY;
                end3 = System.currentTimeMillis() + 4000;
            }
            else if (firstBomb == true) {
                secondBomb = true;
                count++;
                bomb2X = previousPlayerX;
                bomb2Y = previousPlayerY;
                end2 = System.currentTimeMillis() + 4000;
            }
            else {
                firstBomb = true;
                count++;
                bomb1X = previousPlayerX;
                bomb1Y = previousPlayerY;
                end = System.currentTimeMillis() + 4000;
            }
        }
    }

    public boolean firstRemove() {

            firstBomb = false;
            count--;
            return true;
    }

    public boolean secondRemove() {

            secondBomb = false;
            count--;
            return true;
    }

    public boolean thirdRemove() {

            thirdBomb = false;
            count--;
            return true;
    }
}
